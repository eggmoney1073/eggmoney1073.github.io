State Machine

Custom State Machine

몬스터와 스킬이 공통적으로 사용할 수 있는 상태 전환 구조입니다. Enter / Update / Exit를 분리해 각 상태의 책임을 명확하게 나눴습니다.

왜 이렇게 설계했는가?
몬스터와 스킬은 모두 상태에 따라 행동이 바뀌는 객체라서, 각 클래스에 조건문을 계속 늘리는 방식보다 공통 상태머신을 두는 편이 확장에 유리했습니다. 상태 추가나 수정이 생겨도 전체 구조를 흔들지 않고 해당 상태 함수만 교체할 수 있게 설계했습니다.
핵심 포인트
상태별 함수를 Dictionary에 저장하고, ChangeState에서 이전 상태 종료와 다음 상태 진입을 분리한 구조입니다. 
using System;
using System.Collections.Generic;

public class StateMachine<T> where T : Enum
{
    private readonly Dictionary<T, Action> _stateEnter = new Dictionary<T, Action>();
    private readonly Dictionary<T, Action> _stateUpdate = new Dictionary<T, Action>();
    private readonly Dictionary<T, Action> _stateExit = new Dictionary<T, Action>();

    private T _currentState;

    public void SetState(T state, Action enter, Action update, Action exit)
    {
        _stateEnter[state] = enter;
        _stateUpdate[state] = update;
        _stateExit[state] = exit;
    }

    public void ChangeState(T nextState)
    {
        if (_stateExit.ContainsKey(_currentState))
        {
            _stateExit[_currentState]?.Invoke();
        }

        _currentState = nextState;

        if (_stateEnter.ContainsKey(_currentState))
        {
            _stateEnter[_currentState]?.Invoke();
        }
    }

    public void Update()
    {
        if (_stateUpdate.ContainsKey(_currentState))
        {
            _stateUpdate[_currentState]?.Invoke();
        }
    }
}

Reusable Structure

Monster / Skill 공용 사용 구조

몬스터와 스킬이 진영에 따라 완전히 다른 객체가 되지 않도록, Target과 Caster를 기준으로 동작하게 만든 구조입니다.

왜 이렇게 설계했는가?
같은 몬스터가 아군일 수도 있고 적군일 수도 있기 때문에, 진영별로 프리팹과 로직을 따로 만들면 관리 비용이 커집니다. 그래서 누가 쓰는지와 누구를 대상으로 하는지만 바뀌면 같은 구조를 재사용할 수 있게 설계했습니다.
핵심 포인트
스킬 사용 시 Type, Target, Caster를 전달받도록 만들어 하나의 구조를 양쪽 진영에서 공용으로 사용합니다. 
public enum SkillType
{
    Slash,
    Bite
}

public class SkillManager : MonoBehaviour
{
    private Dictionary<SkillType, Queue<Skill>> _skillPool;

    public void ActiveSkill(SkillType type, Transform target, Monster caster)
    {
        Skill skill = GetSkill(type);
        skill.StartSkill(target, caster);
    }

    private Skill GetSkill(SkillType type)
    {
        if (_skillPool[type].Count > 0)
        {
            return _skillPool[type].Dequeue();
        }

        return CreateSkill(type);
    }

    public void FinishSkill(SkillType type, Skill skill)
    {
        _skillPool[type].Enqueue(skill);
    }
}

Optimization

Object Pooling

전투 중 반복 생성되는 몬스터, 스킬, 투사체, 이펙트를 재사용하기 위한 구조입니다.

왜 이렇게 설계했는가?
전투 중 오브젝트를 계속 생성하고 삭제하면 프레임 드랍과 GC 부담이 커질 수 있습니다. 그래서 자주 쓰는 객체는 미리 생성해 두고 재사용하는 방식으로 런타임 비용을 줄이도록 설계했습니다.
핵심 포인트
Queue에 객체를 저장해 두고, 필요할 때 꺼내 쓰고, 끝나면 다시 반환하는 단순하고 확장성 좋은 구조입니다. 
using System;
using System.Collections.Generic;

public class ObjectPool<T>
{
    private readonly Queue<T> _pool = new Queue<T>();
    private readonly Func<T> _createFunc;

    public ObjectPool(Func<T> createFunc, int preloadCount)
    {
        _createFunc = createFunc;

        for (int i = 0; i < preloadCount; i++)
        {
            _pool.Enqueue(_createFunc());
        }
    }

    public T Get()
    {
        if (_pool.Count > 0)
        {
            return _pool.Dequeue();
        }

        return _createFunc();
    }

    public void Set(T item)
    {
        _pool.Enqueue(item);
    }
}

Data Driven

JSON 테이블 로드 구조

몬스터 스탯, 레벨 성장값, 스킬 정보 같은 기획 데이터를 코드 밖에서 관리하기 위한 구조입니다.

왜 이렇게 설계했는가?
밸런스 수치가 자주 바뀌는 프로젝트에서 값을 코드에 직접 적으면 수정 비용이 커집니다. 그래서 데이터를 외부 파일로 분리해 수치 조정과 데이터 추가를 쉽게 만들도록 설계했습니다.
핵심 포인트
JSON 파일을 읽어 테이블 배열로 바꾼 뒤, 타입이나 레벨 기준으로 참조하는 데이터 기반 구조입니다. 
using UnityEngine;

[System.Serializable]
public class MonsterTableData
{
    public int type;
    public float hpMultiplier;
    public float attackMultiplier;
    public int skillIndex;
}

public class MonsterTableLoader : MonoBehaviour
{
    [SerializeField] private TextAsset _monsterTableJson;
    private MonsterTableData[] _monsterTable;

    private void Awake()
    {
        _monsterTable = JsonHelper.FromJson<MonsterTableData>(_monsterTableJson.text);
    }

    public MonsterTableData GetData(int type)
    {
        return _monsterTable[type];
    }
}
