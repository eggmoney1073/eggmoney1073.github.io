Collision

Grid 기반 충돌 판정

투사체와 적의 충돌을 전체 비교가 아닌 Grid 셀 기반으로 판정하는 구조입니다.

왜 이렇게 설계했는가?
적과 투사체 수가 늘어나는 디펜스 게임에서 모든 엔티티를 전수 비교하면 비용이 빠르게 커집니다. 그래서 먼저 Grid에 엔티티를 배치하고, 충돌 검사 시 현재 셀과 주변 셀만 탐색하도록 설계해 충돌 판정 범위를 줄였습니다.
핵심 포인트
GridBuild 단계에서 맵을 채우고, Collision 단계에서 주변 3x3 셀만 검사하는 구조가 핵심입니다. 
using Unity.Burst;
using Unity.Collections;
using Unity.Entities;
using Unity.Mathematics;
using Unity.Transforms;

[BurstCompile]
[WithAll(typeof(ProjectileTag))]
public partial struct Collision_Job : IJobEntity
{
    public int _gridWidth;
    public float _cellSize;

    [ReadOnly]
    public NativeParallelMultiHashMap<int, Entity> _gridMap;

    [ReadOnly]
    public ComponentLookup<LocalTransform> _transformLookup;
    [ReadOnly]
    public ComponentLookup<Collision> _radiusLookup;

    [NativeDisableParallelForRestriction]
    public BufferLookup<HitEnemyBufferElement> _hitBufferLookup;

    [NativeDisableParallelForRestriction]
    public BufferLookup<Damaged> _damagedBufferLookup;

    public void Execute(Entity projectileEntity, in LocalTransform transform, in Collision collider, in ProjectileDamage projectileDamage)
    {
        float3 position = transform.Position;

        int centerX = (int)math.floor(position.x / _cellSize) + (_gridWidth / 2);
        int centerY = (int)math.floor(position.z / _cellSize) + (_gridWidth / 2);

        int offsetY = -1;
        while (offsetY <= 1)
        {
            int offsetX = -1;
            while (offsetX <= 1)
            {
                int axisX = centerX + offsetX;
                int axisY = centerY + offsetY;
                int index = axisX + _gridWidth * axisY;

                NativeParallelMultiHashMapIterator<int> iterator;
                Entity other;

                bool found = _gridMap.TryGetFirstValue(index, out other, out iterator);

                if (found)
                {
                    do
                    {
                        float distanceSq = math.lengthsq(position - _transformLookup[other].Position);
                        float sumRadius = collider.radius + _radiusLookup[other].radius;

                        if (distanceSq <= sumRadius * sumRadius)
                        {
                            DynamicBuffer<HitEnemyBufferElement> hitBuffer = _hitBufferLookup[projectileEntity];
                            DynamicBuffer<Damaged> damagedBuffer = _damagedBufferLookup[other];

                            hitBuffer.Add(new HitEnemyBufferElement { enemy = other });
                            damagedBuffer.Add(new Damaged { damage = projectileDamage.damage });
                        }
                    }
                    while (_gridMap.TryGetNextValue(out other, ref iterator));
                }

                offsetX++;
            }

            offsetY++;
        }
    }
}

Fire System

무기 발사와 투사체 생성

TimeScale을 반영해 발사 쿨다운을 누적하고, 조건이 맞으면 ECS에서 투사체를 생성하는 발사 시스템입니다.

왜 이렇게 설계했는가?
발사 로직을 MonoBehaviour에 두면 ECS 엔티티 흐름과 분리되어 관리 포인트가 늘어납니다. 그래서 무기 상태, 발사 간격, 데미지, 관통 수치까지 ECS 데이터로 처리하고, 발사 순간에 필요한 컴포넌트만 붙여 투사체를 생성하도록 설계했습니다.
핵심 포인트
scaledDeltaTime을 사용해 시간 배속 영향을 반영하고, 발사 시 방향, 데미지, 관통 정보, 피격 버퍼를 함께 세팅하는 구조입니다. 
using Unity.Burst;
using Unity.Entities;
using Unity.Mathematics;
using Unity.Transforms;

[BurstCompile]
[UpdateInGroup(typeof(SpawnSystemGroup))]
partial struct MainFire_System : ISystem
{
    EntityQuery _projectileQuery;
    int _spawnIndex;

    public void OnCreate(ref SystemState state)
    {
        _projectileQuery = state.GetEntityQuery(ComponentType.ReadOnly<ProjectileTag>());
        _spawnIndex = 0;
    }

    public void OnUpdate(ref SystemState state)
    {
        EntityCommandBuffer ecb = new EntityCommandBuffer(Unity.Collections.Allocator.Temp);
        TimeScaleData timeData = SystemAPI.GetSingleton<TimeScaleData>();
        float deltaTime = timeData.scaledDeltaTime;

        foreach ((RefRW<FireRate> fireData, RefRW<ProjectilePrefab> projectile, RefRO<WeaponDamage> weaponDamage, RefRO<WeaponPierce> pierce, Entity entity)
            in SystemAPI.Query<RefRW<FireRate>, RefRW<ProjectilePrefab>, RefRO<WeaponDamage>, RefRO<WeaponPierce>>().WithAll<WeaponTag>().WithEntityAccess())
        {
            fireData.ValueRW.timer += deltaTime;

            if (fireData.ValueRO.timer >= fireData.ValueRO.coolDown)
            {
                fireData.ValueRW.timer = 0f;

                Entity projectileEntity = ecb.Instantiate(projectile.ValueRO.projectileEntity);

                float3 dir = AimDirectionBridge.AimDirection;
                float angle = math.atan2(dir.y, dir.x);
                float3 spawnPosition = new float3(0, 0, _spawnIndex * -0.001f);
                _spawnIndex++;

                ecb.SetComponent(projectileEntity, LocalTransform.FromPositionRotationScale(spawnPosition, quaternion.RotateZ(angle), 1f));
                ecb.AddComponent(projectileEntity, new Direction { moveDirection = dir });
                ecb.AddComponent(projectileEntity, new ProjectileDamage { damage = weaponDamage.ValueRO.damage });
                ecb.AddComponent(projectileEntity, new PierceData { maxPierceCount = pierce.ValueRO.pierceCount });
                ecb.AddBuffer<HitEnemyBufferElement>(projectileEntity);
            }
        }

        ecb.Playback(state.EntityManager);
        ecb.Dispose();
    }
}

Lifecycle

LiveTag 기반 생명주기 관리

체력이 0 이하가 된 적을 즉시 Destroy하지 않고, LiveTag를 비활성화한 뒤 삭제 시스템에서 정리하는 구조입니다.

왜 이렇게 설계했는가?
판정 중간에 즉시 삭제하면 같은 프레임 안에서 다른 시스템과 충돌할 수 있습니다. 그래서 죽음 판정과 실제 삭제를 분리해, 먼저 비활성 상태로 전환하고 마지막 단계에서 안전하게 제거하도록 설계했습니다.
핵심 포인트
LiveTag를 끄는 단계와, WithDisabled 조건으로 엔티티를 정리하는 단계를 나눈 구조입니다. 
using Unity.Entities;

public struct LiveTag : IComponentData, IEnableableComponent { }

[WithAll(typeof(EnemyTag))]
partial struct EnemyDead_System : ISystem
{
    public void OnUpdate(ref SystemState state)
    {
        foreach ((RefRO<EnemyHealth> enemyHealth, EnabledRefRW<LiveTag> liveTag, Entity entity)
            in SystemAPI.Query<RefRO<EnemyHealth>, EnabledRefRW<LiveTag>>().WithEntityAccess())
        {
            if (enemyHealth.ValueRO.health <= 0)
            {
                liveTag.ValueRW = false;

                RefRW<GameProgress> progress = SystemAPI.GetSingletonRW<GameProgress>();
                progress.ValueRW.kill += 1;
            }
        }
    }
}

public partial struct EntityDestroy_Job : IJobEntity

Time Management

TimeScale 관리 시스템

일반 DeltaTime이 아니라 게임 시간 배속을 반영한 scaledDeltaTime을 만들어 전체 시스템이 같은 기준으로 시간을 사용하도록 만든 구조입니다.

왜 이렇게 설계했는가?
일시정지, 슬로우 모션, 배속 같은 기능을 later 넣을 때, 각 시스템이 제각각 시간을 계산하면 관리가 어려워집니다. 그래서 공용 TimeScaleData를 두고 모든 시스템이 같은 시간 기준을 보도록 설계했습니다.
핵심 포인트
timeScale, scaledDeltaTime, unscaledDeltaTime을 함께 관리하는 공용 시간 구조입니다. 
using Unity.Burst;
using Unity.Entities;

[BurstCompile]
[UpdateInGroup(typeof(ManageSystemGroup))]
partial struct TimeManager_System : ISystem
{
    [BurstCompile]
    public void OnCreate(ref SystemState state)
    {
        if (SystemAPI.HasSingleton<TimeScaleData>() == false)
        {
            Entity entity = state.EntityManager.CreateEntity();

            state.EntityManager.AddComponentData(entity, new TimeScaleData
            {
                timeScale = 1f,
                scaledDeltaTime = 0f,
                time = 0f,
                unscaledDeltaTime = 0f
            });
        }
    }

    [BurstCompile]
    public void OnUpdate(ref SystemState state)
    {
        float deltaTime = SystemAPI.Time.DeltaTime;
        RefRW<TimeScaleData> timeData = SystemAPI.GetSingletonRW<TimeScaleData>();

        timeData.ValueRW.scaledDeltaTime = deltaTime * timeData.ValueRO.timeScale;
        timeData.ValueRW.unscaledDeltaTime = deltaTime;
        timeData.ValueRW.time += deltaTime * timeData.ValueRO.timeScale;
    }
}
